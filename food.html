import { getFirestore, collection, addDoc, deleteDoc, doc, onSnapshot, serverTimestamp } from 'firebase/firestore';

 

// --- Firebase Initialization ---

// Safely handle environment variables

let firebaseConfig = {};

try {

  if (typeof __firebase_config !== 'undefined') {

    firebaseConfig = JSON.parse(__firebase_config);

  }

} catch (e) {

  console.warn("Firebase config parsing error, using default/empty config.", e);

}

 

const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);

const appId = typeof __app_id !== 'undefined' ? __app_id : 'bento-app-default';

 

// --- Menu Data (Parsed from the provided image) ---

  {

    needsOption: true, // Needs Rice or Noodle selection

    ]

];

 

export default function App() {

  // Firebase Auth State

  const [user, setUser] = useState(null);

 

  // App State

  const [orders, setOrders] = useState([]);

  const [userName, setUserName] = useState(localStorage.getItem('bento_username') || '');

  const [isNameSet, setIsNameSet] = useState(!!localStorage.getItem('bento_username'));

  const [activeCategory, setActiveCategory] = useState('主餐');

  const [isCartOpen, setIsCartOpen] = useState(false);

  const [errorMsg, setErrorMsg] = useState('');

  const [successMsg, setSuccessMsg] = useState('');

 

  // Modal State for Main Meal options

  const [selectedItemForOption, setSelectedItemForOption] = useState(null);

  const [mealOption, setMealOption] = useState('飯'); // 飯 or 麵

 

  // 1. Initialize Auth

  useEffect(() => {

    const initAuth = async () => {

      try {

        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

          await signInWithCustomToken(auth, __initial_auth_token);

        } else {

          await signInAnonymously(auth);

        }

      } catch (err) {

        console.error("Auth Error:", err);

        setErrorMsg("無法連接至認證伺服器，請稍後再試。");

      }

    };

    initAuth();

 

    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {

      setUser(currentUser);

    });

    return () => unsubscribe();

  }, []);

 

  // 2. Fetch Orders (Public Group Data)

  useEffect(() => {

    if (!user) return;

 

    // Use Public Data path for group ordering app (Rule 1)

    const ordersRef = collection(db, 'artifacts', appId, 'public', 'data', 'group_orders');

   

    const unsubscribe = onSnapshot(ordersRef, (snapshot) => {

      const fetchedOrders = snapshot.docs.map(doc => ({

        id: doc.id,

        ...doc.data()

      }));

      // Sort in memory (Rule 2)

      fetchedOrders.sort((a, b) => {

        const timeA = a.timestamp?.toMillis?.() || 0;

        const timeB = b.timestamp?.toMillis?.() || 0;

        return timeA - timeB; // Oldest first

      });

      setOrders(fetchedOrders);

    }, (err) => {

      console.error("Firestore Subscribe Error:", err);

      setErrorMsg("資料庫連線異常，無法取得最新訂單。");

    });

 

    return () => unsubscribe();

  }, [user]);

 

  // Auto-hide success message

  useEffect(() => {

    if (successMsg) {

      const timer = setTimeout(() => setSuccessMsg(''), 3000);

      return () => clearTimeout(timer);

    }

  }, [successMsg]);

 

  // Handle setting username

  const handleSetName = (e) => {

    e.preventDefault();

    if (userName.trim()) {

      localStorage.setItem('bento_username', userName.trim());

      setIsNameSet(true);

    }

  };

 

  // Add Item to Cart

  const handleAddToCart = async (item, option = null) => {

    if (!user) {

      setErrorMsg("請等待系統連線...");

      return;

    }

    if (!isNameSet) {

      alert("請先輸入您的稱呼！");

      return;

    }

 

    try {

      const ordersRef = collection(db, 'artifacts', appId, 'public', 'data', 'group_orders');

      await addDoc(ordersRef, {

        uid: user.uid,

        userName: userName.trim(),

        category: activeCategory,

        itemName: item.name,

        price: item.price,

        option: item.fixedOption || option, // '飯', '麵', '冬粉' etc

        timestamp: serverTimestamp()

      });

      setSelectedItemForOption(null); // close modal

    } catch (err) {

      console.error("Add Doc Error:", err);

      setErrorMsg("新增訂單失敗，請檢查網路連線。");

    }

  };

 

  // Delete Item from Cart

  const handleDeleteItem = async (orderId, orderUid) => {

    // Only allow deleting own orders

    if (user.uid !== orderUid) {

      alert("您只能刪除自己的餐點喔！");

      return;

    }

    try {

      const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'group_orders', orderId);

      await deleteDoc(docRef);

    } catch (err) {

      console.error("Delete Doc Error:", err);

      setErrorMsg("刪除訂單失敗。");

    }

  };

 

  // Pre-add check for Options (飯/麵)

  const preAddToCart = (item, categoryData) => {

    if (categoryData.needsOption && !item.fixedOption) {

      setSelectedItemForOption(item);

      setMealOption('飯'); // default

    } else {

      handleAddToCart(item);

    }

  };

 

  // Compute Cart Totals & Discounts

  const { totalAmount, discountAmount, groupedOrders } = useMemo(() => {

    let rawTotal = 0;

    let mainMealCount = 0;

    let misoSoupCount = 0;

    const grouped = {};

 

    orders.forEach(order => {

      rawTotal += order.price;

     

      // Count for Miso Soup Discount

      if (order.category === '主餐') mainMealCount++;

      if (order.itemName === '味噌湯') misoSoupCount++;

 

      // Group by user

      if (!grouped[order.userName]) {

        grouped[order.userName] = { items: [], subtotal: 0, uid: order.uid };

      }

      grouped[order.userName].items.push(order);

      grouped[order.userName].subtotal += order.price;

    });

 

    // Calculate discount: 10 NTD off for each Miso Soup paired with a Main Meal

    const applicableDiscounts = Math.min(mainMealCount, misoSoupCount);

    const calculatedDiscount = applicableDiscounts * 10;

 

    return {

      totalAmount: rawTotal - calculatedDiscount,

      discountAmount: calculatedDiscount,

      groupedOrders: grouped

    };

  }, [orders]);

 

  const handleConfirmOrder = () => {

    const myOrders = orders.filter(o => o.uid === user?.uid);

    if (myOrders.length === 0) {

      setErrorMsg("您還沒有點選任何餐點喔！");

      return;

    }

    setSuccessMsg("訂單已確認！您的餐點已加入揪團明細。");

    setIsCartOpen(false);

  };

 

  // --- Render Helpers ---

  const activeCategoryData = MENU_DATA.find(c => c.category === activeCategory);

 

  if (!isNameSet) {

    return (

      <div className="min-h-screen bg-orange-50 flex items-center justify-center p-4">

        <div className="bg-white p-8 rounded-2xl shadow-xl max-w-md w-full text-center">

          <div className="w-16 h-16 bg-orange-100 text-orange-600 rounded-full flex items-center justify-center mx-auto mb-4">

            <User className="w-8 h-8" />

          </div>

          <h1 className="text-2xl font-bold text-gray-800 mb-2">梁社漢排骨 - 揪團訂餐</h1>

          <p className="text-gray-500 mb-6">請輸入您的稱呼，方便大家知道是誰點的餐喔！</p>

          <form onSubmit={handleSetName} className="space-y-4">

            <input

              type="text"

              required

              value={userName}

              onChange={(e) => setUserName(e.target.value)}

              placeholder="例如：王小明、業務部-Claire"

              className="w-full px-4 py-3 rounded-xl border border-gray-200 focus:outline-none focus:ring-2 focus:ring-orange-500 transition-shadow"

            />

            <button

              type="submit"

              className="w-full bg-orange-500 hover:bg-orange-600 text-white font-semibold py-3 rounded-xl transition-colors shadow-md hover:shadow-lg"

            >

              開始點餐

            </button>

          </form>

        </div>

      </div>

    );

  }

 

  return (

    <div className="min-h-screen bg-gray-50 text-gray-800 font-sans flex flex-col md:flex-row">

     

      {/* Mobile Header / Cart Toggle */}

      <div className="md:hidden bg-white shadow-sm p-4 flex justify-between items-center sticky top-0 z-20">

        <div>

          <h1 className="font-bold text-lg text-orange-600">梁社漢排骨</h1>

          <p className="text-xs text-gray-500">訂餐人：{userName}</p>

        </div>

        <button

          onClick={() => setIsCartOpen(!isCartOpen)}

          className="relative p-2 bg-orange-100 text-orange-600 rounded-full"

        >

          <ShoppingCart className="w-6 h-6" />

          {orders.length > 0 && (

            <span className="absolute -top-1 -right-1 bg-red-500 text-white text-[10px] w-5 h-5 flex items-center justify-center rounded-full border-2 border-white">

              {orders.length}

            </span>

          )}

        </button>

      </div>

 

      {/* Success Message Toast */}

      {successMsg && (

        <div className="fixed top-4 left-1/2 -translate-x-1/2 z-50 bg-green-100 border border-green-400 text-green-700 px-4 py-3 rounded shadow-lg flex items-center z-[60]">

          <CheckCircle2 className="w-5 h-5 mr-2" />

          <span>{successMsg}</span>

          <button onClick={() => setSuccessMsg('')} className="ml-4 font-bold hover:text-green-900">&times;</button>

        </div>

      )}

 

      {/* Error Message Toast */}

      {errorMsg && (

        <div className="fixed top-4 left-1/2 -translate-x-1/2 z-50 bg-red-100 border border-red-400 text-red-700 px-4 py-3 rounded flex items-center shadow-lg z-[60]">

          <AlertCircle className="w-5 h-5 mr-2" />

          <span>{errorMsg}</span>

          <button onClick={() => setErrorMsg('')} className="ml-4 font-bold">&times;</button>

        </div>

      )}

 

      {/* Main Menu Section */}

      <div className={`flex-1 flex flex-col h-screen overflow-hidden ${isCartOpen ? 'hidden md:flex' : 'flex'}`}>

        {/* Desktop Header */}

        <div className="hidden md:flex bg-white shadow-sm p-6 justify-between items-center z-10">

          <div>

            <h1 className="text-2xl font-bold text-orange-600">梁社漢排骨 - 揪團訂餐系統</h1>

            <p className="text-sm text-gray-500 mt-1 flex items-center">

              <User className="w-4 h-4 mr-1" /> 目前登入：<span className="font-semibold text-gray-700 ml-1">{userName}</span>

              <button onClick={() => setIsNameSet(false)} className="ml-3 text-orange-500 hover:text-orange-700 underline text-xs">更換名稱</button>

            </p>

          </div>

        </div>

 

        {/* Category Navigation */}

        <div className="bg-white border-b border-gray-200 overflow-x-auto hide-scrollbar">

          <div className="flex p-2 md:p-4 space-x-2 md:space-x-4 min-w-max">

            {MENU_DATA.map((cat) => (

              <button

                key={cat.category}

                onClick={() => setActiveCategory(cat.category)}

                className={`flex items-center px-4 py-2 md:px-6 md:py-3 rounded-full font-medium transition-colors ${

                  activeCategory === cat.category

                    ? 'bg-orange-500 text-white shadow-md'

                    : 'bg-gray-100 text-gray-600 hover:bg-gray-200'

                }`}

              >

                <span className="mr-2">{cat.icon}</span>

                {cat.category}

              </button>

            ))}

          </div>

        </div>

 

        {/* Menu Items Grid */}

        <div className="flex-1 overflow-y-auto p-4 md:p-6 bg-gray-50">

          {activeCategory === '湯品 / 飲料' && (

            <div className="mb-4 bg-blue-50 text-blue-700 p-3 rounded-lg flex items-start text-sm">

              <Info className="w-5 h-5 mr-2 shrink-0 mt-0.5" />

              <p>好康提醒：味噌湯搭配「主餐」折抵 10 元！（系統將於結帳區自動幫您折抵）</p>

            </div>

          )}

 

          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 md:gap-6">

            {activeCategoryData.items.map((item, idx) => (

              <div

                key={idx}

                className="bg-white rounded-2xl p-4 md:p-5 shadow-sm hover:shadow-md border border-gray-100 transition-shadow flex flex-col justify-between"

              >

                <div>

                  <div className="flex justify-between items-start mb-2">

                    <h3 className="font-bold text-lg text-gray-800 leading-tight">

                      {item.name}

                      {item.new && <span className="ml-2 inline-block px-1.5 py-0.5 bg-red-100 text-red-600 text-[10px] rounded border border-red-200 font-bold align-middle">新品</span>}

                      {item.spicy && <span className="ml-2 text-red-500 text-sm align-middle" title="會辣">🌶️</span>}

                    </h3>

                    <span className="font-bold text-orange-600 text-lg">${item.price}</span>

                  </div>

                  {(item.note || item.discountNote) && (

                    <p className="text-xs text-gray-500 mt-1">

                      {item.note && <span className="bg-gray-100 px-2 py-1 rounded mr-2">{item.note}</span>}

                      {item.discountNote && <span className="text-orange-500">{item.discountNote}</span>}

                    </p>

                  )}

                </div>

               

                <button

                  onClick={() => preAddToCart(item, activeCategoryData)}

                  className="mt-4 w-full flex items-center justify-center py-2.5 bg-orange-50 hover:bg-orange-100 text-orange-600 font-semibold rounded-xl transition-colors"

                >

                  <Plus className="w-4 h-4 mr-1" /> 新增至訂單

                </button>

              </div>

            ))}

          </div>

         

          <div className="h-20 md:h-0"></div> {/* Bottom padding for mobile cart toggle */}

        </div>

      </div>

 

      {/* Sidebar Cart / Group Orders */}

      <div className={`w-full md:w-80 lg:w-96 bg-white shadow-xl flex flex-col h-screen fixed md:static top-0 right-0 z-30 transform transition-transform duration-300 ease-in-out ${isCartOpen ? 'translate-x-0' : 'translate-x-full md:translate-x-0'}`}>

        <div className="p-4 md:p-6 border-b border-gray-100 flex justify-between items-center bg-orange-50 md:bg-white">

          <h2 className="text-xl font-bold text-gray-800 flex items-center">

            <ShoppingCart className="w-5 h-5 mr-2 text-orange-500" />

            揪團總明細

          </h2>

          <button

            onClick={() => setIsCartOpen(false)}

            className="md:hidden text-gray-500 hover:text-gray-700"

          >

            ✕

          </button>

        </div>

 

        <div className="flex-1 overflow-y-auto p-4 md:p-6">

          {orders.length === 0 ? (

            <div className="h-full flex flex-col items-center justify-center text-gray-400">

              <ShoppingCart className="w-16 h-16 mb-4 opacity-20" />

              <p>目前還沒有人點餐喔～</p>

            </div>

          ) : (

            <div className="space-y-6">

              {Object.entries(groupedOrders).map(([person, data]) => (

                <div key={person} className="bg-gray-50 rounded-xl p-4 border border-gray-100">

                  <div className="flex justify-between items-center mb-3 pb-2 border-b border-gray-200">

                    <span className="font-bold text-gray-700 flex items-center">

                      <User className="w-4 h-4 mr-1 text-gray-400" />

                      {person}

                      {person === userName && <span className="ml-2 text-[10px] bg-orange-100 text-orange-600 px-1.5 py-0.5 rounded">您</span>}

                    </span>

                    <span className="font-semibold text-gray-600">${data.subtotal}</span>

                  </div>

                  <ul className="space-y-2">

                    {data.items.map(item => (

                      <li key={item.id} className="flex justify-between items-start text-sm group">

                        <div className="flex-1">

                          <span className="text-gray-800">{item.itemName}</span>

                          {item.option && <span className="ml-1 text-xs text-gray-500">({item.option})</span>}

                        </div>

                        <div className="flex items-center ml-2">

                          <span className="text-gray-600 mr-2">${item.price}</span>

                          {/* Only show delete if it belongs to current user */}

                          {item.uid === user?.uid ? (

                            <button

                              onClick={() => handleDeleteItem(item.id, item.uid)}

                              className="text-red-400 hover:text-red-600 transition-colors p-1"

                              title="刪除"

                            >

                              <Trash2 className="w-4 h-4" />

                            </button>

                          ) : (

                            <div className="w-6"></div> /* Spacer to align prices */

                          )}

                        </div>

                      </li>

                    ))}

                  </ul>

                </div>

              ))}

            </div>

          )}

        </div>

 

        {/* Totals & Checkout Footer */}

        <div className="border-t border-gray-200 p-4 md:p-6 bg-white">

          {discountAmount > 0 && (

            <div className="flex justify-between text-sm text-green-600 mb-2 items-center">

              <span className="flex items-center"><CheckCircle2 className="w-4 h-4 mr-1" /> 味噌湯搭配折扣</span>

              <span>-${discountAmount}</span>

            </div>

          )}

          <div className="flex justify-between items-end mb-4">

            <span className="text-gray-600 font-medium">總計金額</span>

            <span className="text-3xl font-bold text-orange-600">${totalAmount}</span>

          </div>

          <button

            onClick={handleConfirmOrder}

            className="w-full bg-gray-800 hover:bg-gray-900 text-white font-bold py-3.5 rounded-xl transition-colors shadow-lg flex justify-center items-center"

          >

            確認送出訂單

          </button>

        </div>

      </div>

 

      {/* Modal: Select Rice or Noodle */}

      {selectedItemForOption && (

        <div className="fixed inset-0 bg-black/50 backdrop-blur-sm flex items-center justify-center z-50 p-4">

          <div className="bg-white rounded-2xl shadow-xl w-full max-w-sm overflow-hidden animate-in fade-in zoom-in-95 duration-200">

            <div className="p-6">

              <h3 className="text-lg font-bold text-gray-900 mb-1">{selectedItemForOption.name}</h3>

              <p className="text-sm text-gray-500 mb-5">請選擇搭配主食</p>

             

              <div className="grid grid-cols-2 gap-3 mb-6">

                <button

                  onClick={() => setMealOption('飯')}
